Find the nth term of Geometric Progression:
Input: A, R, N
Output: Nth term

#Example:
Input: A=2, R=3, N=5
Output: 162

#include <iostream>
using namespace std;

int main()
{
 int a,r,n,x,s,y=1,z=0;

  cout<<"Enter a,r and n "<<endl;
  cin>>a>>r>>n;      //Input data

  z=n-1;
for(int i=1;i<=z;i++)   //to calculate (r)^n
{
  x=y*r;
  y=x;
}

  s=x*a;    //Formula
 cout <<s; 
  
}
