Expected Time Complexity: O(log(min(A, B)))
Expected Auxiliary Space: O(1)

Code:-
#include <iostream>
using namespace std;

int main() 
{
  int n1, n2;

  cout << "Enter two numbers: ";
  cin >> n1 >> n2;           //Input
  
  while(n1 != n2) {     //Condition to find GCD
    if(n1 > n2)
      n1=n1-n2;
    else
      n2=n2-n1;
  }

  cout<< n1;  //Output
  
  return 0;
}
