Time Complexity: O(log(min(A, B)))
 Space Complexity: O(1)

code:
int gcd(int A, int B) 
{
  while(A != B)  //Condition to find GCD
  {             
    if(A > B)
      A=A-B;
    else
      B=B-A;
  }
  return A;  //Output      
}
