#include <iostream>
using namespace std;

int main()
{	
    long long int n;
    int a;

      cout<<"enter number"<<endl;   //get input form user
      cin>>n;
		
	while(n>0)
	 {
	  a=n%10;       //find remainder
	  cout<<a;     //will display remainder
	  n=n/10;      
	 }	    
					
}
